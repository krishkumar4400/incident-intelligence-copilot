const fs = require("fs");

// ---------- CONFIG ----------
const SERVICES = [
  "frontend",
  "auth-service",
  "payment-service",
  "order-service",
  "inventory-service",
  "notification-service",
  "database"
];

const INCIDENT_COUNT = 5;
const ALERTS_PER_INCIDENT = 50;

// ---------- UTILS ----------
function randomInt(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

function getRandomService() {
  return SERVICES[randomInt(0, SERVICES.length - 1)];
}

function now() {
  return Date.now();
}

// ---------- DATA HOLDERS ----------
let alerts = [];
let deployments = [];
let incidents = [];

// ---------- GENERATE DEPLOYMENTS ----------
function generateDeployment(service, time) {
  return {
    id: "dep_" + Math.random().toString(36).substr(2, 9),
    service,
    version: "v" + randomInt(1, 5) + "." + randomInt(0, 9),
    timestamp: time
  };
}

// ---------- GENERATE ALERT ----------
function generateAlert(service, baseTime, incidentId) {
  const metrics = ["latency", "cpu", "memory", "db_connections"];

  return {
    id: "alert_" + Math.random().toString(36).substr(2, 9),
    incident_id: incidentId,
    service,
    metric: metrics[randomInt(0, metrics.length - 1)],
    value: randomInt(500, 1500),
    severity: ["low", "medium", "high"][randomInt(0, 2)],
    timestamp: baseTime + randomInt(0, 60000) // within 1 min
  };
}

// ---------- INCIDENT SIMULATION ----------
function generateIncident(i) {
  const incidentId = "inc_" + i;

  const rootService = getRandomService();
  const incidentTime = now() + i * 100000;

  // 1. Deployment before incident
  const deployment = generateDeployment(rootService, incidentTime - 60000);
  deployments.push(deployment);

  // 2. Alerts (root cause + cascading)
  for (let j = 0; j < ALERTS_PER_INCIDENT; j++) {
    let service;

    if (j < 10) {
      // Root service alerts
      service = rootService;
    } else {
      // Cascading alerts
      service = getRandomService();
    }

    alerts.push(generateAlert(service, incidentTime, incidentId));
  }

  incidents.push({
    id: incidentId,
    root_service: rootService,
    deployment_id: deployment.id,
    timestamp: incidentTime
  });
}

// ---------- MAIN ----------
function generate() {
  for (let i = 0; i < INCIDENT_COUNT; i++) {
    generateIncident(i);
  }

  fs.writeFileSync("data/alerts.json", JSON.stringify(alerts, null, 2));
  fs.writeFileSync("data/deployments.json", JSON.stringify(deployments, null, 2));
  fs.writeFileSync("data/incidents.json", JSON.stringify(incidents, null, 2));

  console.log("✅ Dataset generated successfully!");
}

generate();
